   179тЖТ   179тЖТfunction deliverableSummary(deliverables: Deliverable[]): string {
   180тЖТ   180тЖТ  if (deliverables.length === 0) return "No deliverables";
   181тЖТ   181тЖТ  const counts: Record<string, number> = {};
   182тЖТ   182тЖТ  for (const d of deliverables) {
   183тЖТ   183тЖТ    const k = deliverableStatusKey(d.status);
   184тЖТ   184тЖТ    counts[k] = (counts[k] ?? 0) + 1;
   185тЖТ   185тЖТ  }
   186тЖТ   186тЖТ  const order = ["pending", "in_progress", "review", "delivered"];
   187тЖТ   187тЖТ  const labels: Record<string, string> = {
   188тЖТ   188тЖТ    pending: "pending",
   189тЖТ   189тЖТ    in_progress: "in progress",
   190тЖТ   190тЖТ    review: "in review",
   191тЖТ   191тЖТ    delivered: "delivered",
   192тЖТ   192тЖТ  };
   193тЖТ   193тЖТ  return order
   194тЖТ   194тЖТ    .filter((k) => counts[k])
   195тЖТ   195тЖТ    .map((k) => `${counts[k]} ${labels[k]}`)
   196тЖТ   196тЖТ    .join(", ");
   197тЖТ   197тЖТ}
