    40‚Üíasync function buildContextSummary() {
    41‚Üí  const now = new Date();
    42‚Üí
    43‚Üí  const [opportunities, invoices, expenses, tasks, projects, paymentRequests] = await Promise.all([
    44‚Üí    db.opportunity.findMany({ include: { account: true } }),
    45‚Üí    db.invoice.findMany(),
    46‚Üí    db.expense.findMany(),
    47‚Üí    db.task.findMany({ orderBy: { dueDate: "asc" } }),
    48‚Üí    db.project.findMany(),
    49‚Üí    db.paymentRequest.findMany(),
    50‚Üí  ]);
    51‚Üí
    52‚Üí  const openOpps = opportunities.filter((o) => !["WON", "LOST"].includes(o.stage));
    53‚Üí  const pipelineValue = openOpps.reduce((s, o) => s + o.value, 0);
    54‚Üí
    55‚Üí  const outstandingInvoices = invoices.filter(
    56‚Üí    (i) => i.status === "OVERDUE" || i.status === "PARTIAL" || i.status === "SENT"
    57‚Üí  );
    58‚Üí  const outstandingAmount = outstandingInvoices.reduce((s, i) => s + (i.amount - i.amountPaid), 0);
    59‚Üí  const overdueInvoices = invoices.filter((i) => i.status === "OVERDUE");
    60‚Üí  const overdueAmount = overdueInvoices.reduce((s, i) => s + (i.amount - i.amountPaid), 0);
    61‚Üí
    62‚Üí  const totalRevenue = invoices.reduce((s, i) => s + i.amountPaid, 0);
    63‚Üí  const totalExpenses = expenses.reduce((s, e) => s + e.amount, 0);
    64‚Üí  const cashPosition = totalRevenue - totalExpenses;
    65‚Üí
    66‚Üí  const pendingApprovals = paymentRequests.filter((p) => p.status === "PENDING").length;
    67‚Üí  const openTasks = tasks.filter((t) => t.status !== "DONE").length;
    68‚Üí  const overdueTasks = tasks.filter(
    69‚Üí    (t) => t.status !== "DONE" && t.dueDate && new Date(t.dueDate) < now
    70‚Üí  ).length;
    71‚Üí  const activeProjects = projects.filter(
    72‚Üí    (p) => p.status === "PLANNING" || p.status === "CONFIRMED" || p.status === "IN_PROGRESS"
    73‚Üí  ).length;
    74‚Üí  const distractions = tasks.filter((t) => t.isDistraction && t.status !== "DONE").length;
    75‚Üí
    76‚Üí  // top priorities ‚Äî top 5 due soon/overdue by priority
    77‚Üí  const order: Record<string, number> = { URGENT: 0, HIGH: 1, MEDIUM: 2, LOW: 3 };
    78‚Üí  const topPriorities = tasks
    79‚Üí    .filter((t) => t.status !== "DONE" && t.dueDate && new Date(t.dueDate) <= new Date(now.getTime() + 86400000))
    80‚Üí    .sort((a, b) => (order[a.priority] ?? 2) - (order[b.priority] ?? 2))
    81‚Üí    .slice(0, 5)
    82‚Üí    .map((t) => t.title);
    83‚Üí
    84‚Üí  // upcoming deadlines within 7 days (tasks + project event dates + unpaid invoice due dates)
    85‚Üí  const upcomingDeadlines = [
    86‚Üí    ...tasks
    87‚Üí      .filter((t) => t.status !== "DONE" && t.dueDate)
    88‚Üí      .map((t) => ({ title: t.title, due: t.dueDate! })),
    89‚Üí    ...projects
    90‚Üí      .filter((p) => p.eventDate)
    91‚Üí      .map((p) => ({ title: p.name, due: p.eventDate! })),
    92‚Üí    ...invoices
    93‚Üí      .filter((i) => i.dueDate && i.status !== "PAID")
    94‚Üí      .map((i) => ({ title: `${i.code ?? "Invoice"}`, due: i.dueDate! })),
    95‚Üí  ]
    96‚Üí    .filter((x) => {
    97‚Üí      const d = new Date(x.due);
    98‚Üí      return d >= new Date(now.getTime() - 86400000) && d <= new Date(now.getTime() + 7 * 86400000);
    99‚Üí    })
   100‚Üí    .sort((a, b) => new Date(a.due).getTime() - new Date(b.due).getTime())
   101‚Üí    .slice(0, 8)
   102‚Üí    .map((x) => x.title);
   103‚Üí
   104‚Üí  return {
   105‚Üí    pipelineValue,
   106‚Üí    outstandingAmount,
   107‚Üí    overdueAmount,
   108‚Üí    pendingApprovals,
   109‚Üí    openTasks,
   110‚Üí    overdueTasks,
   111‚Üí    activeProjects,
   112‚Üí    distractions,
   113‚Üí    cashPosition,
   114‚Üí    topPriorities,
   115‚Üí    upcomingDeadlines,
   116‚Üí  };
   117‚Üí}
