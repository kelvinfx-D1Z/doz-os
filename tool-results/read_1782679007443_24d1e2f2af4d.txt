   237тЖТ   237тЖТ  const start = new Date(now.getFullYear(), now.getMonth(), now.getDate());

   264тЖТ   264тЖТ
   265тЖТ   265тЖТfunction CascadeRow({
   266тЖТ   266тЖТ  goal,
   267тЖТ   267тЖТ  level,
   268тЖТ   268тЖТ}: {
   269тЖТ   269тЖТ  goal: {
   270тЖТ   270тЖТ    id: string;
   271тЖТ   271тЖТ    title: string;
   272тЖТ   272тЖТ    status: string;
   273тЖТ   273тЖТ    progress: number;
   274тЖТ   274тЖТ    dueDate: string;
   275тЖТ   275тЖТ    owner: { name: string };
   276тЖТ   276тЖТ    quarter?: string | null;
   277тЖТ   277тЖТ  };
   278тЖТ   278тЖТ  level: number;
   279тЖТ   279тЖТ}) {
   280тЖТ   280тЖТ  const days = daysUntil(goal.dueDate);
   281тЖТ   281тЖТ  const isOverdue = days < 0 && goal.status !== "ACHIEVED";
   282тЖТ   282тЖТ  const isAchieved = goal.status === "ACHIEVED";
   283тЖТ   283тЖТ  const levelLabel = LEVEL_LABEL[level] ?? "Goal";
   284тЖТ   284тЖТ
   285тЖТ   285тЖТ  return (
   286тЖТ   286тЖТ    <div
   287тЖТ   287тЖТ      className={cn(
   288тЖТ   288тЖТ        level > 0 && "border-l-2 border-primary/40 pl-4",

   290тЖТ   290тЖТ        level === 2 && "ml-4",
   291тЖТ   291тЖТ        level === 3 && "ml-8",

   572тЖТ   572тЖТfunction HealthDonut({
   573тЖТ   573тЖТ  active,
   574тЖТ   574тЖТ  achieved,
   575тЖТ   575тЖТ  missed,
   576тЖТ   576тЖТ}: {

   593тЖТ   593тЖТ    <div className="flex items-center gap-5">
